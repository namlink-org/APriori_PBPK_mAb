# Write xlsx to a fresh temp file then copy to target.
# This bypasses Windows file-lock errors when a file is open in Excel.
safe_save_xlsx <- function(wb, target_path) {
  tmp <- tempfile(fileext = ".xlsx")
  on.exit(try(file.remove(tmp), silent = TRUE))
  openxlsx::saveWorkbook(wb, tmp, overwrite = TRUE)
  if (file.exists(target_path))
    try(file.remove(target_path), silent = TRUE)
  file.copy(tmp, target_path, overwrite = TRUE)
  invisible(target_path)
}

# ── Applications.xlsx ─────────────────────────────────────────────────────────
# One sheet per scenario. Container path confirmed from V2 manual file.
# All routes treated as 1-min bolus approximation (as per V2 convention).

APP_CONTAINER <- "Events|IV_Bolus|Application_1|ProtocolSchemaItem"

wb_app <- createWorkbook()

for (i in seq_len(nrow(scenarios))) {
  s  <- scenarios[i, ]
  df <- data.frame(
    `Container Path`     = APP_CONTAINER,
    `Parameter Name`     = "DosePerBodyWeight",
    Value                = s$Dose,
    Units                = "mg/kg",
    `Infusion time`      = 1,
    `Infusion time unit` = "min",
    check.names = FALSE, stringsAsFactors = FALSE
  )
  addWorksheet(wb_app, s$Sheet.name)
  writeData(wb_app, s$Sheet.name, df)
  setColWidths(wb_app, s$Sheet.name, cols = 1:6,
               widths = c(55, 22, 8, 8, 14, 18))
}

# ── ModelParameters.xlsx ──────────────────────────────────────────────────────
# One sheet per Molecule x Species, plus a Global sheet (empty).
# Confirmed structure from V2 manual file: 6 parameters per sheet, short paths.

PARAM_NAMES <- c("Kd (FcRn) in endosomal space", "Kd", "koff",
                  "kint", "kdeg", "Reference concentration")

mol_species <- scenarios %>% distinct(Molecule, Species.label)

wb_mp <- createWorkbook()

# Global sheet — required by esqlabsR; empty = no project-wide overrides
addWorksheet(wb_mp, "Global")
writeData(wb_mp, "Global",
          data.frame(`Container Path` = NA_character_,
                     `Parameter Name` = NA_character_,
                     Value = NA_real_, Units = NA_character_,
                     check.names = FALSE))

for (i in seq_len(nrow(mol_species))) {
  mol  <- mol_species$Molecule[i]
  spec <- mol_species$Species.label[i]

  # Use human parameters file for human scenarios, monkey file for monkey
  param_source <- if (spec == "Human") params_long_human else params_long

  mol_rows <- param_source %>%
    filter(Drug.Name == mol, Parameter.Name %in% PARAM_NAMES) %>%
    mutate(Parameter.Name = factor(Parameter.Name, levels = PARAM_NAMES)) %>%
    arrange(Parameter.Name) %>%
    transmute(
      `Container Path` = container_map[as.character(Parameter.Name)],
      `Parameter Name` = as.character(Parameter.Name),
      Value            = as.numeric(Value),
      Units            = unit_map[as.character(Parameter.Name)]
    )

  sheet_nm <- paste0(mol, "_", spec)
  addWorksheet(wb_mp, sheet_nm)
  writeData(wb_mp, sheet_nm, mol_rows)
  setColWidths(wb_mp, sheet_nm, cols = 1:4, widths = c(28, 32, 12, 10))
}

mp_path <- file.path(confdir, "ModelParameters.xlsx")
safe_save_xlsx(wb_mp, mp_path)
message("ModelParameters.xlsx written: ", nrow(mol_species),
        " molecule x species sheets + Global")
