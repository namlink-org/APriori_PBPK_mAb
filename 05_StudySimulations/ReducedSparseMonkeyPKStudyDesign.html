Purpose

This workflow implements a two-gate decision analysis for using a priori PBPK and a reduced sparse monkey PK study to support monoclonal antibody PK characterization before human trial initiation.

Gate 1: a priori adequacy. The a priori monkey prediction is compared with observed monkey PK exposure from the retrospective dataset. A molecule triggers a reduced monkey PK study if at least one a priori monkey AUC fold-error is outside the prespecified 2-fold interval.

Gate 2: reduced sparse-study adequacy. For triggered molecules only, the top-down monkey model is treated as the retrospective reference for true monkey PK. The analysis asks whether a reduced sparse monkey PK study with small sample size can recover the top-down population exposure within 2-fold.

Manuscript note: this analysis intentionally does not bias-correct the a priori model. Bias is part of Gate 1. Once Gate 1 fails, the reduced study design is evaluated against the observed-informed/top-down monkey reference, not against the a priori model.

make_lognormal <- function(u, typical, cv) {
  sdlog <- sqrt(log1p(cv^2))
  qlnorm(u, meanlog = log(typical), sdlog = sdlog)
}

random_quantiles <- function(n, p) {
  matrix(runif(n * p), nrow = n, ncol = p)
}

conc_nm_to_ug_ml <- function(conc_nm, mw_g_mol) {
  as.numeric(conc_nm) * as.numeric(mw_g_mol) / 1e6
}

dose_mgkg_to_nmolkg <- function(dose_mgkg, mw_g_mol) {
  as.numeric(dose_mgkg) * 1e6 / as.numeric(mw_g_mol)
}

linup_logdown_auc <- function(time_h, concentration) {
  keep <- is.finite(time_h) & is.finite(concentration) & concentration >= 0
  time_h <- time_h[keep]
  concentration <- concentration[keep]
  ord <- order(time_h)
  time_h <- time_h[ord]
  concentration <- concentration[ord]
  if (length(time_h) < 2L) return(NA_real_)

  dt <- diff(time_h)
  c1 <- head(concentration, -1)
  c2 <- tail(concentration, -1)
  declining <- c2 < c1 & c1 > 0 & c2 > 0
  intervals <- dt * (c1 + c2) / 2
  intervals[declining] <-
    dt[declining] * (c1[declining] - c2[declining]) /
    log(c1[declining] / c2[declining])
  sum(intervals)
}

geometric_mean <- function(x) {
  x <- x[is.finite(x) & x > 0]
  exp(mean(log(x)))
}

within_fold <- function(estimate, reference, fold = FOLD_ACCEPTANCE) {
  ratio <- estimate / reference
  is.finite(ratio) && ratio >= 1 / fold && ratio <= fold
}

Build top-down reference population

The top-down monkey model is treated as the retrospective reference for true monkey PK among Gate 1-triggered molecules. This avoids circularity: the a priori model determines whether a reduced study is needed, but the reduced study is evaluated against an observed-informed monkey reference.

Simulate sparse top-down monkey PK

Code 
qss_rxode_model <- rxode2::rxode({
  Ltot <- centr / vc
  Cp <- peri / vp
  free_drug <- 0.5 * ((Ltot - total_target - kss) +
    sqrt((Ltot - total_target - kss)^2 + 4 * kss * Ltot + 1e-12))
  complex <- Ltot - free_drug
  free_target <- total_target - complex

  d/dt(centr) = -cl * free_drug - q * free_drug + q * Cp -
    kint * complex * vc
  d/dt(peri) = q * free_drug - q * Cp
  d/dt(total_target) = ksyn - kdeg * free_target - kint * complex
})

simulate_topdown_profile <- function(parameter_row, dose_mgkg, times_h) {
  theta <- c(
    cl = parameter_row$CL_L_h_kg,
    vc = parameter_row$Vc_L_kg,
    q = parameter_row$Q_L_h_kg,
    vp = parameter_row$Vp_L_kg,
    r0 = parameter_row$Rtot_nM,
    kdeg = parameter_row$kdeg_h,
    kint = parameter_row$kint_h,
    kss = parameter_row$Kss_nM,
    ksyn = parameter_row$Rtot_nM * parameter_row$kdeg_h
  )

  event_table <- data.frame(
    id = parameter_row$IndividualId,
    time = c(0, times_h),
    amt = c(
      dose_mgkg_to_nmolkg(dose_mgkg, parameter_row$MW),
      rep(NA_real_, length(times_h))
    ),
    evid = c(1, rep(0, length(times_h))),
    cmt = "centr"
  )

  sol <- rxode2::rxSolve(
    qss_rxode_model,
    params = theta,
    events = event_table,
    inits = c(centr = 0, peri = 0, total_target = theta[["r0"]]),
    rtol = 1e-6,
    atol = 1e-9,
    hmin = 1e-12,
    maxsteps = 100000,
    returnType = "data.frame"
  ) %>%
    as_tibble()

  if ("evid" %in% names(sol)) {
    sol <- sol %>% filter(evid == 0 | is.na(evid))
  }

  sol %>%
    transmute(
      Scenario_name = paste(
        parameter_row$Drug, "Monkey", paste0("Dose", dose_mgkg), sep = "_"
      ),
      Drug = parameter_row$Drug,
      IndividualId = as.character(parameter_row$IndividualId),
      Dose_mgkg = dose_mgkg,
      Time_h = as.numeric(time),
      Time_day = Time_h / 24,
      Concentration_ug_mL = conc_nm_to_ug_ml(
        pmax(as.numeric(Ltot), 0),
        parameter_row$MW
      )
    )
}

Reduced sparse-study adequacy

For each molecule, dose, and sample size, reduced studies are simulated by resampling n animals from the top-down reference population and calculating the study geometric mean sparse AUC. The reference is the geometric mean sparse AUC from the full top-down virtual population. A simulated reduced study is successful if the study geometric mean is within 2-fold of the full-population reference.

simulate_single_dose_reduced_study <- function(auc_values, n,
                                               n_sim = N_SIM,
                                               fold = FOLD_ACCEPTANCE) {
  reference <- geometric_mean(auc_values)
  tibble(Replicate = seq_len(n_sim)) %>%
    mutate(
      Study_AUC_geomean_ug_h_mL = map_dbl(
        Replicate,
        ~ geometric_mean(sample(auc_values, n, replace = TRUE))
      ),
      Fold_error = Study_AUC_geomean_ug_h_mL / reference,
      Success_within_2fold =
        Fold_error >= 1 / fold & Fold_error <= fold
    )
}

single_dose_reduced_study_replicates <- crossing(
  topdown_sparse_auc %>% distinct(Drug, Dose_mgkg),
  N_per_group = N_PER_GROUP
) %>%
  mutate(
    Replicates = pmap(
      list(Drug, Dose_mgkg, N_per_group),
      ~ {
        auc_values <- topdown_sparse_auc %>%
          filter(Drug == ..1, Dose_mgkg == ..2) %>%
          pull(Sparse_AUC_ug_h_mL)

Two- and three-dose reduced study designs

The dose-design analysis is stricter than the single-dose analysis. A simulated design replicate is successful only if every selected dose group is within 2-fold of its corresponding full-population top-down sparse AUC reference.

Code 
simulate_design_replicate <- function(drug, doses, n_by_dose,
                                      fold = FOLD_ACCEPTANCE) {
  dose_success <- logical(length(doses))

  for (i in seq_along(doses)) {
    auc_values <- topdown_sparse_auc %>%
      filter(Drug == drug, Dose_mgkg == doses[i]) %>%
      pull(Sparse_AUC_ug_h_mL)
    reference <- geometric_mean(auc_values)
    study_gmean <- geometric_mean(sample(auc_values, n_by_dose[i], replace = TRUE))
    dose_success[i] <- within_fold(study_gmean, reference, fold)
  }

  all(dose_success)
}

simulate_design_success <- function(drug, doses, n_by_dose,
                                    n_sim = N_SIM) {
  mean(replicate(
    n_sim,
    simulate_design_replicate(drug, doses, n_by_dose)
  ))
}

Suggested manuscript text:

We used a two-gate decision framework. First, a priori PBPK predictions were screened against observed monkey exposure. Molecules with at least one a priori AUC fold-error outside the 2-fold interval were considered cases where a reduced monkey PK study would be triggered. Second, for these triggered molecules, the observed-informed top-down monkey model was used as the retrospective reference for true monkey PK. Reduced sparse PK studies were simulated by sampling small groups of animals from the top-down virtual population and comparing the study geometric mean AUC with the full top-down population geometric mean AUC. This design separates model bias from sampling adequacy: failure of Gate 1 reflects limitations of the a priori prediction, whereas Gate 2 evaluates whether a reduced sparse monkey study is sufficient once in vivo PK characterization is needed.

In this framework, increasing sample size is not expected to correct systematic a priori model bias. Instead, the reduced study simulation quantifies whether a targeted sparse monkey PK study can recover the observed-informed monkey exposure estimate with adequate precision, thereby supporting decisions about whether reduced in vivo PK data are sufficient for human trial initiation.
